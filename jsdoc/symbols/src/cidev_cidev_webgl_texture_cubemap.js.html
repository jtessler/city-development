1 /**
  2  * @fileoverview Cubemap texture subclass.
  3  *
  4  * @author joseph@cs.utexas.edu (Joe Tessler)
  5  */
  6 
  7 goog.provide('cidev.webgl.texture.Cubemap');
  8 
  9 goog.require('cidev.webgl.texture.Texture');
 10 
 11 goog.require('goog.uri.utils');
 12 goog.require('goog.webgl');
 13 
 14 /**
 15  * @param {!cidev.webgl.Context} context The current WebGL context.
 16  * @param {string} path URL path to a folder of 6 images named
 17  *     "{pos,neg}{x,y,z}.png".
 18  * @param {number=} opt_unit The optional unit index, where the maximum amount
 19  *     of texture units is hardware dependent. Defaults to 0.
 20  * @constructor
 21  * @extends {cidev.webgl.texture.Texture}
 22  */
 23 cidev.webgl.texture.Cubemap = function(context, path, opt_unit) {
 24   goog.base(this, context, opt_unit);
 25 
 26   var gl = context.gl;
 27   gl.texParameteri(goog.webgl.TEXTURE_CUBE_MAP, goog.webgl.TEXTURE_MAG_FILTER,
 28                    goog.webgl.LINEAR);
 29   gl.texParameteri(goog.webgl.TEXTURE_CUBE_MAP, goog.webgl.TEXTURE_MIN_FILTER,
 30                    goog.webgl.LINEAR);
 31   gl.texParameteri(goog.webgl.TEXTURE_CUBE_MAP, goog.webgl.TEXTURE_WRAP_S,
 32                    goog.webgl.CLAMP_TO_EDGE);
 33   gl.texParameteri(goog.webgl.TEXTURE_CUBE_MAP, goog.webgl.TEXTURE_WRAP_T,
 34                    goog.webgl.CLAMP_TO_EDGE);
 35 
 36   for (var i = 0; i < 6; i++) {
 37     var image = new Image();
 38     image.src = goog.uri.utils.appendPath(
 39         path, cidev.webgl.texture.Cubemap.IMAGES[i]);
 40     image.onload = goog.bind(
 41         function(target, img) {
 42             this.activate();
 43             this.context.gl.texImage2D(
 44                 target,
 45                 0,
 46                 goog.webgl.RGBA,
 47                 goog.webgl.RGBA,
 48                 goog.webgl.UNSIGNED_BYTE,
 49                 img);
 50         }, this, goog.webgl.TEXTURE_CUBE_MAP_POSITIVE_X + i, image);
 51   }
 52 };
 53 goog.inherits(cidev.webgl.texture.Cubemap, cidev.webgl.texture.Texture);
 54 
 55 /**
 56  * @inheritDoc
 57  */
 58 cidev.webgl.texture.Cubemap.prototype.bindTexture = function() {
 59   this.context.gl.bindTexture(goog.webgl.TEXTURE_CUBE_MAP, this.texture);
 60 };
 61 
 62 /**
 63  * The filenames for each cube map face.
 64  * @type {Array.<string>}
 65  * @const
 66  */
 67 cidev.webgl.texture.Cubemap.IMAGES = [
 68     'posx.png',
 69     'negx.png',
 70     'posy.png',
 71     'negy.png',
 72     'posz.png',
 73     'negz.png'
 74 ];
