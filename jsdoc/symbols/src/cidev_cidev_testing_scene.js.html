 21 /**
 22  * @param {!Element} canvas The canvas container for the scene.
 23  * @constructor
 24  * @implements {goog.fx.anim.Animated}
 25  */
 26 cidev.testing.Scene = function(canvas) {
 27   /** @type {!cidev.webgl.Context} */
 28   this.context = new cidev.webgl.Context(canvas);
 29 
 30   /** @type {!cidev.webgl.Camera} */
 31   this.camera = new cidev.webgl.Camera(this.context);
 32   this.camera.pos[0] = 5;
 33   this.camera.pos[1] = 1;
 34 
 35   /** @type {!goog.vec.Mat4.Float32} */
 36   this.matrix = goog.vec.Mat4.createFloat32Identity();
 37 
 38   /** @type {!cidev.webgl.shader.Program} */
 39   this.simple = new cidev.webgl.shader.DiffuseSpecular(
 40       this.context,
 41       this.matrix,
 42       goog.vec.Vec3.createFloat32FromValues(-40, 40, 40));
 43 
 44   /** @type {!cidev.webgl.texture.Texture} */
 45   this.concrete = new cidev.webgl.texture.Texture2D(
 46       this.context, 'textures/concrete.jpg', 0);
 47 
 48   /** @type {!cidev.webgl.Mesh} */
 49   this.plant = new cidev.webgl.Mesh(this.context, 'power_plant.obj');
 50 
 51   /** @type {!cidev.webgl.shader.Program} */
 52   this.skybox = new cidev.webgl.shader.Skybox(this.context, this.matrix);
 53 
 54   /** @type {!cidev.webgl.texture.Texture} */
 55   this.cubemap = new cidev.webgl.texture.Cubemap(
 56       this.context, 'textures/cubemap', 1);
 57 
 58   /** @type {!cidev.webgl.Mesh} */
 59   this.cube = new cidev.webgl.Mesh(this.context, 'cube.obj');
 60 
 61   /** @type {!cidev.webgl.texture.Texture} */
 62   this.grass = new cidev.webgl.texture.Texture2D(
 63       this.context, 'textures/grass.jpg', 3);
 64 
 65   /** @type {!cidev.webgl.Mesh} */
 66   this.ground = new cidev.webgl.Mesh(this.context, 'ground.obj');
 67 
 68   /** @type {!cidev.webgl.texture.Texture} */
 69   this.facade = new cidev.webgl.texture.Texture2D(
 70       this.context, 'textures/building_flip.jpg', 4);
 71 
 72   /** @type {!cidev.webgl.Mesh} */
 73   this.building = new cidev.webgl.Mesh(
 74       this.context, 'residential_building.obj');
 75 
 76   /** @type {number} */
 77   this.lastTime = 0;
 78 };
 79 
 80 /**
 81  * @inheritDoc
 82  */
 83 cidev.testing.Scene.prototype.onAnimationFrame = function(now) {
 84   this.context.gl.clear(
 85       goog.webgl.COLOR_BUFFER_BIT | goog.webgl.DEPTH_BUFFER_BIT);
 86 
 87   if (this.lastTime > 0) {
 88     this.camera.update(now - this.lastTime);
 89   }
 90   this.lastTime = now;
 91 
 92   this.simple.activate();
 93   goog.vec.Mat4.makeTranslate(this.matrix, 5, 0, 5);
 94   for (var i = 0; i < 3; i++) {
 95     this.simple.render(this.building, this.camera, this.facade);
 96     goog.vec.Mat4.translate(this.matrix, 0, 2, 0);
 97   }
 98 
 99   goog.vec.Mat4.makeTranslate(this.matrix, 2, 0, 2);
100   this.simple.render(this.plant, this.camera, this.concrete);
101 
102   for (var r = 0; r < 10; r++) {
103     for (var c = 0; c < 10; c++) {
104       goog.vec.Mat4.makeTranslate(this.matrix, r, 0, c);
105       this.simple.render(this.ground, this.camera, this.grass);
106     }
107   }
108 
109   this.skybox.activate();
110   goog.vec.Mat4.makeScale(this.matrix, 100, 100, 100);
111   goog.vec.Mat4.translate(this.matrix, -0.5, -0.5, -0.5);
112   this.skybox.render(this.cube, this.camera, this.cubemap);
113 };
