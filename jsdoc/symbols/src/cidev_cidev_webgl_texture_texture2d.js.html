1 /**
  2  * @fileoverview A generic 2D texture wrapper.
  3  *
  4  * @author joseph@cs.utexas.edu (Joe Tessler)
  5  */
  6 
  7 goog.provide('cidev.webgl.texture.Texture2D');
  8 
  9 goog.require('cidev.webgl.texture.Texture');
 10 
 11 goog.require('goog.webgl');
 12 
 13 /**
 14  * @param {!cidev.webgl.Context} context The current WebGL context.
 15  * @param {string} url URL path to the texture image.
 16  * @param {number=} opt_unit The optional unit index, where the maximum amount
 17  *     of texture units is hardware dependent. Defaults to 0.
 18  * @constructor
 19  * @extends {cidev.webgl.texture.Texture}
 20  */
 21 cidev.webgl.texture.Texture2D = function(context, url, opt_unit) {
 22   goog.base(this, context, opt_unit);
 23 
 24   var gl = context.gl;
 25   gl.texParameteri(goog.webgl.TEXTURE_2D, goog.webgl.TEXTURE_MAG_FILTER,
 26                    goog.webgl.LINEAR);
 27   gl.texParameteri(goog.webgl.TEXTURE_2D, goog.webgl.TEXTURE_MIN_FILTER,
 28                    goog.webgl.LINEAR_MIPMAP_NEAREST);
 29 
 30   var image = new Image();
 31   image.src = url;
 32   image.onload = goog.bind(
 33       function() {
 34           this.activate();
 35           this.context.gl.texImage2D(
 36               goog.webgl.TEXTURE_2D,
 37               0,
 38               goog.webgl.RGBA,
 39               goog.webgl.RGBA,
 40               goog.webgl.UNSIGNED_BYTE,
 41               image);
 42           this.context.gl.generateMipmap(goog.webgl.TEXTURE_2D);
 43       }, this);
 44 };
 45 goog.inherits(cidev.webgl.texture.Texture2D, cidev.webgl.texture.Texture);
 46 
 47 /**
 48  * @inheritDoc
 49  */
 50 cidev.webgl.texture.Texture2D.prototype.bindTexture = function() {
 51   this.context.gl.bindTexture(goog.webgl.TEXTURE_2D, this.texture);
 52 };
