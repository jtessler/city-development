1 /**
  2  * TODO(joseph): Make a more robust database system.
  3  * @fileoverview A simple, static database class, storing model information in
  4  * memory.

  7  */
  8 
  9 goog.provide('cidev.database');
 10 
 11 goog.require('goog.object');
 12 
 13 /**
 14  * The in-memory database, storing building models by UID.
 15  * @type {Object.<string, !cidev.model.Building>}
 16  * @private
 17  */
 18 cidev.database.db_ = {};
 19 
 20 /**
 21  * Adds a building to the database and assigns it an ID.
 22  * @param {!cidev.model.Building} building The model to store.
 23  */
 24 cidev.database.add = function(building) {
 25   building.id = 'in-memory-' + goog.getUid(building);
 26   goog.object.add(cidev.database.db_, building.id, building);
 27 };
 28 
 29 /**
 30  * Returns the building stored in the database.
 31  * @param {string} id The ID of the building in the database.
 32  * @return {!cidev.model.Building|undefined} The stored model.
 33  */
 34 cidev.database.get = function(id) {
 35   return goog.object.get(cidev.database.db_, id);
 36 };
 37 
 38 /**
 39  * Returns all buildings stored in the database.
 40  * @return {Array.<!cidev.model.Building>} All stored buildings.
 41  */
 42 cidev.database.getAll = function() {
 43   return goog.object.getValues(cidev.database.db_);
 44 };
 45 
 46 /**
 47  * Updates the stored building in the database.
 48  * NOTE: This is a no-op for this in-memory database.
 49  * @param {!cidev.model.Building} building The model to store.
 50  */
 51 cidev.database.update = function(building) {
 52   // This is a no-op for an in-memory database.
 53 };
 54 
 55 /**
 56  * Removes the given building from the database.
 57  * @param {!cidev.model.Building} building The model to remove.
 58  */
 59 cidev.database.remove = function(building) {
 60   goog.object.remove(cidev.database.db_, building.id);
 61 };
