 15 /**
 16  * Initializes the WebGL context, event handlers, and draws the scene.
 17  */
 18 cidev.base.init = function() {
 19   var canvas = goog.dom.getElement('canvas');
 20   if (goog.isDefAndNotNull(canvas)) {
 21 
 22     cidev.controller.setupContainer(goog.dom.getElement('container'));
 23     var scene = new cidev.Scene(canvas);
 24 
 25     // TODO(joseph): Find a better way to wait for everything to load.
 26     setTimeout(goog.partial(goog.fx.anim.registerAnimation, scene), 500);
 27 
 28   } else {
 29     throw Error('canvas does not exist');
 30   }
 31 };
 32 goog.exportSymbol('init', cidev.base.init);
