  7 goog.provide('cidev.webgl.Context');
  8 
  9 goog.require('goog.vec.Mat4');
 10 goog.require('goog.webgl');
 11 
 12 /**
 13  * Constructs a new WebGL context wrapper.
 14  * @param {!Element} canvas The Canvas container for WebGL.
 15  * @constructor
 16  */
 17 cidev.webgl.Context = function(canvas) {
 18   /** @type {!Element} */
 19   this.canvas = canvas;
 20 
 21   /** @type {!WebGLRenderingContext} */
 22   this.gl = cidev.webgl.Context.createWebGLContext(canvas);
 23 
 24   var gl = this.gl;
 25   gl.clearColor(0.0, 0.0, 0.0, 1.0);
 26   gl.enable(goog.webgl.DEPTH_TEST);
 27 
 28   /**
 29    * The field of view along the y (vertical) axis in radians.
 30    * @type {number}
 31    */
 32   this.fov = Math.PI / 4;
 33 
 34   /**
 35    * The distance to the near clipping plane.
 36    * @type {number}
 37    */
 38   this.near = 1.0;
 39 
 40   /**
 41    * The distance to the far clipping plane.
 42    * @type {number}
 43    */
 44   this.far = 200.0;
 45 
 46   /**
 47    * The projection matrix.
 48    * @type {!goog.vec.Mat4.Float32}
 49    */
 50   this.projMatrix = goog.vec.Mat4.createFloat32();
 51 
 52   this.update();
 53 };
 54 
 55 /**
 56  * Updates the viewport and projection matrix based on the canvas's size and
 57  * other user-mutable properties.
 58  */
 59 cidev.webgl.Context.prototype.update = function() {
 60   this.gl.viewport(0, 0, this.canvas.width, this.canvas.height);
 61   goog.vec.Mat4.makePerspective(
 62       this.projMatrix,
 63       this.fov,
 64       this.canvas.width / this.canvas.height,
 65       this.near,
 66       this.far);
 67 };

 69 /**
 70  * Creates a WebGL context in a given canvas in a browser-independent way.
 71  * @param {!Element} canvas The canvas tag from which to get context.
 72  * @return {!WebGLRenderingContext} The created context.
 73  */
 74 cidev.webgl.Context.createWebGLContext = function(canvas) {
 75   if (!window.WebGLRenderingContext) {
 76     throw Error('unsupported browser');
 77   }
 78 
 79   var names = ['webgl', 'experimental-webgl', 'webkit-3d', 'moz-webgl'];
 80   var context = null;
 81   for (var i = 0; goog.isNull(context) && i < names.length; i++) {
 82     try {
 83       context = canvas.getContext(names[i]);
 84     } catch (e) { /* Do nothing. */ }
 85   }
 86 
 87   if (goog.isNull(context)) {
 88     throw Error('unsupported hardware');
 89   }
 90 
 91   return context;
 92 };
