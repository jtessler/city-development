 15 /**
 16  * The representing any 3D object, exposing binds for each vertex attribute.
 17  * @param {!cidev.webgl.Context} context The current WebGL context.
 18  * @param {string} filename The OBJ filename to import as a wrapped mesh.
 19  * @constructor
 20  */
 21 cidev.webgl.Mesh = function(context, filename) {
 22   this.context = context;
 23 
 24   var gl = context.gl;
 25   this.vertexBuffer_ = gl.createBuffer();
 26   this.textureUVBuffer_ = gl.createBuffer();
 27   this.normalBuffer_ = gl.createBuffer();
 28   this.indexBuffer_ = gl.createBuffer();
 29 
 30   var obj = cidev.generated.obj[filename];
 31   if (!goog.isDef(obj)) {
 32     throw Error(filename + ' does not exist');
 33   }
 34 
 35   this.bindVertexBuffer();
 36   gl.bufferData(goog.webgl.ARRAY_BUFFER, obj.vertices, goog.webgl.STATIC_DRAW);
 37 
 38   this.bindTextureUVBuffer();
 39   gl.bufferData(goog.webgl.ARRAY_BUFFER, obj.textureUVs,
 40       goog.webgl.STATIC_DRAW);
 41 
 42   this.bindNormalBuffer();
 43   gl.bufferData(goog.webgl.ARRAY_BUFFER, obj.normals, goog.webgl.STATIC_DRAW);
 44 
 45   this.bindIndexBuffer();
 46   gl.bufferData(
 47       goog.webgl.ELEMENT_ARRAY_BUFFER, obj.indices, goog.webgl.STATIC_DRAW);
 48   this.indexCount = obj.indices.length;
 49 
 50   // Assign basic material properties for the user to reassign later.
 51   this.ma = goog.vec.Vec3.createFloat32FromValues(0.4, 0.4, 0.4);
 52   this.md = goog.vec.Vec3.createFloat32FromValues(0.5, 0.5, 0.5);
 53   this.ms = goog.vec.Vec3.createFloat32FromValues(0.7, 0.7, 0.7);
 54   this.shininess = 100;
 55 };
