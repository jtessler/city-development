1 /**
  2  * @fileoverview The textured M.V.P. program superclass.
  3  *
  4  * This class assumes the GLSL shader code includes everything specified in the
  5  * superclasses, plus the following:
  6  *
  7  *   attribute vec2 textureUV;
  8  *
  9  *   uniform sampler2D texture;
 10  *
 11  * @author joseph@cs.utexas.edu (Joe Tessler)
 12  */
 13 
 14 goog.provide('cidev.webgl.shader.TextureProgram');
 15 
 16 goog.require('cidev.webgl.shader.MVPProgram');
 17 
 18 goog.require('goog.webgl');
 19 
 20 /**
 21  * @param {!cidev.webgl.Context} context The WebGL context wrapper.
 22  * @param {string} vs Vertex shader code filename.
 23  * @param {string} fs Fragment shader code filename.
 24  * @param {!goog.vec.Mat4.Float32} matrix Pointer to the scene's model matrix.
 25  * @constructor
 26  * @extends {cidev.webgl.shader.MVPProgram}
 27  */
 28 cidev.webgl.shader.TextureProgram = function(context, vs, fs, matrix) {
 29   goog.base(this, context, vs, fs, matrix);
 30 
 31   var gl = context.gl;
 32 
 33   /**
 34    * Index of the vertex texture UV attribute.
 35    * @type {number}
 36    * @private
 37    */
 38   this.textureUV_ = gl.getAttribLocation(this.program, 'textureUV');
 39   gl.enableVertexAttribArray(this.textureUV_);
 40 
 41   this.texture = gl.getUniformLocation(this.program, 'texture');
 42 };
 43 goog.inherits(cidev.webgl.shader.TextureProgram, cidev.webgl.shader.MVPProgram);
 44 
 45 /**
 46  * Sets up the render call, binding vertex attributes, uniforms, etc.
 47  * @inheritDoc
 48  */
 49 cidev.webgl.shader.TextureProgram.prototype.setupRender = function(
 50     mesh, camera, opt_texture) {
 51   goog.base(this, 'setupRender', mesh, camera, opt_texture);
 52 
 53   mesh.bindTextureUVBuffer();
 54   this.context.gl.vertexAttribPointer(
 55       this.textureUV_, 2, goog.webgl.FLOAT, false, 0, 0);
 56 };
