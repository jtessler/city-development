 15 /**
 16  * The panel container for all property displays and input boxes.

 21 /**
 22  * The select element containing all stored building models.
 23  * @type {!Element}

 28  * @typedef {{building: !cidev.model.Building, input: !Element}}

 30 cidev.view.BuildingInput;

 32 /**
 33  * Initializes DOM elements and event listeners.
 34  * @param {Element} container The DOM element to fill.
 35  */
 36 cidev.view.init = function(container) {
 37   goog.dom.removeChildren(container);
 38 
 39   var selectElement = goog.dom.createDom('select');
 40   goog.events.listen(selectElement, goog.events.EventType.CHANGE,
 41       cidev.controller.switchBuilding, false, selectElement);
 42   container.appendChild(selectElement);
 43   cidev.view.selector = selectElement;
 44   cidev.view.buildingSelector([]); // Starts with no buildings.
 45 
 46   var addPowerPlantElement = goog.dom.createDom('input',
 47       {'type': 'button', 'value': 'Add Power Plant'});
 48   goog.events.listen(addPowerPlantElement, goog.events.EventType.CLICK,
 49       cidev.controller.addPowerPlant, false);
 50   container.appendChild(addPowerPlantElement);
 51 
 52   var addResidentialElement = goog.dom.createDom('input',
 53       {'type': 'button', 'value': 'Add Residential Building'});
 54   goog.events.listen(addResidentialElement, goog.events.EventType.CLICK,
 55       cidev.controller.addResidentialBuilding, false);
 56   container.appendChild(addResidentialElement);
 57 
 58   var addSchoolElement = goog.dom.createDom('input',
 59       {'type': 'button', 'value': 'Add School'});
 60   goog.events.listen(addSchoolElement, goog.events.EventType.CLICK,
 61       cidev.controller.addSchool, false);
 62   container.appendChild(addSchoolElement);
 63 
 64   var propertyPanel = goog.dom.createDom('div');
 65   cidev.view.panel = propertyPanel;
 66   container.appendChild(propertyPanel);
 67 };
 68 
 69 /**
 70  * Updates the select element given a list of building models.
 71  * @param {Array.<!cidev.model.Building>} buildings The stored building models.
 72  */
 73 cidev.view.buildingSelector = function(buildings) {
 74   goog.dom.removeChildren(cidev.view.selector);
 75   if (goog.array.isEmpty(buildings)) {
 76     cidev.view.selector.appendChild(
 77         goog.dom.createDom('option', null, 'Select a building to modify'));
 78   } else {
 79     goog.array.forEach(buildings,
 80         function(building, i, array) {
 81           this.appendChild(goog.dom.createDom('option',
 82               { 'value': building.id },
 83               goog.string.buildString('(', i + 1, ') ', building.getType())));
 84         }, cidev.view.selector);
 85   }
 86 };

 88 /**
 89  * Updates the properties DOM for the given model.
 90  * @param {!cidev.model.Building} building The model to build a panel for.
 91  */
 92 cidev.view.propertyPanel = function(building) {
 93   cidev.view.clearPropertyPanel();
 94   cidev.view.selector.value = building.id; // Update the current selection.
 95 
 96   var xElement = goog.dom.createDom('input',
 97       {'type': 'text', 'value': building.x});
 98   goog.events.listen(xElement, goog.events.EventType.CHANGE,
 99       cidev.controller.updateX, false,
100       {building: building, input: xElement});
101   cidev.view.panel.appendChild(
102       goog.dom.createDom('span', null, 'X-coordinate: '));
103   cidev.view.panel.appendChild(xElement);
104 
105   var yElement = goog.dom.createDom('input',
106       {'type': 'text', 'value': building.y});
107   goog.events.listen(yElement, goog.events.EventType.CHANGE,
108       cidev.controller.updateY, false,
109       {building: building, input: yElement});
110   cidev.view.panel.appendChild(
111       goog.dom.createDom('span', null, 'Y-coordinate: '));
112   cidev.view.panel.appendChild(yElement);
113 
114   switch (building.getType()) {
115     case cidev.model.BuildingType.RESIDENTIAL:
116       var floorCountElement = goog.dom.createDom('input',
117           {'type': 'text', 'value': building.floorCount});
118       goog.events.listen(floorCountElement, goog.events.EventType.CHANGE,
119           cidev.controller.updateFloorCount, false,
120           {building: building, input: floorCountElement});
121       cidev.view.panel.appendChild(
122           goog.dom.createDom('span', null, 'Number of floors: '));
123       cidev.view.panel.appendChild(floorCountElement);
124       break;
125     case cidev.model.BuildingType.POWER_PLANT:
126       var radiusElement = goog.dom.createDom('input',
127           {'type': 'text', 'value': building.radius});
128       goog.events.listen(radiusElement, goog.events.EventType.CHANGE,
129           cidev.controller.updateRadius, false,
130           {building: building, input: radiusElement});
131       cidev.view.panel.appendChild(
132           goog.dom.createDom('span', null, 'Radius: '));
133       cidev.view.panel.appendChild(radiusElement);
134   }
135 
136   var removeElement = goog.dom.createDom('input',
137       {'type': 'button', 'value': 'Remove Building'});
138   goog.events.listen(removeElement, goog.events.EventType.CLICK,
139       cidev.controller.removeBuilding, false, building);
140   cidev.view.panel.appendChild(removeElement);
141 };
142 
143 /**
144  * Clears the properties DOM.
145  */
146 cidev.view.clearPropertyPanel = function() {
147   // Remove all previously attached listeners.
148   goog.array.forEach(goog.dom.getChildren(cidev.view.panel),
149       function(child, i, array) { goog.events.removeAll(child); });
150   goog.dom.removeChildren(cidev.view.panel);
151 };
