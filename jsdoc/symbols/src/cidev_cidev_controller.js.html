 50 /**
 51  * @param {!cidev.model.Building} building The newly constructed building.
 52  * @private
 53  */
 54 cidev.controller.addBuilding_ = function(building) {
 55   cidev.database.add(building);
 56   cidev.view.buildingSelector(cidev.database.getAll());
 57   cidev.view.propertyPanel(building);
 58 };
 59 
 60 /**
 61  * @this {!Element} The select element.
 62  * @param {*} e The handled event.
 63  */
 64 cidev.controller.switchBuilding = function(e) {
 65   var building = cidev.database.get(this.value);
 66   if (goog.isDef(building)) {
 67     cidev.view.propertyPanel(building);
 68   }
 69 };
 70 
 71 /**
 72  * @this {!cidev.model.Building} The building being removed.
 73  * @param {*} e The handled event.
 74  */
 75 cidev.controller.removeBuilding = function(e) {
 76   cidev.database.remove(this);
 77   var buildings = cidev.database.getAll();
 78   cidev.view.buildingSelector(buildings);
 79   if (goog.array.isEmpty(buildings)) {
 80     cidev.view.clearPropertyPanel();
 81   } else {
 82     cidev.view.propertyPanel(buildings[0]);
 83   }
 84 };

 86 /**
 87  * @this {!cidev.view.BuildingInput} The model to modify with the input.
 88  * @param {*} e The handled event.
 89  */
 90 cidev.controller.updateX = function(e) {
 91   if (goog.string.isNumeric(this.input.value)) {
 92     this.building.x = goog.string.parseInt(this.input.value);
 93   }
