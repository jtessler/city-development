  8 goog.provide('cidev.webgl.texture.Texture');

 10 goog.require('goog.webgl');
 11 
 12 /**
 13  * The superclass representing any texture, wrapping binds and the unit index.
 14  * @param {!cidev.webgl.Context} context The current WebGL context.
 15  * @param {number=} opt_unit The optional unit index, where the maximum amount
 16  *     of texture units is hardware dependent. Defaults to 0.
 17  * @constructor
 18  */
 19 cidev.webgl.texture.Texture = function(context, opt_unit) {
 20   this.context = context;
 21   this.unit = goog.isDef(opt_unit) ? opt_unit : 0;
 22 
 23   var gl = context.gl;
 24   var unitSize = gl.getParameter(goog.webgl.MAX_COMBINED_TEXTURE_IMAGE_UNITS);
 25   if (this.unit < 0 || (goog.isNumber(unitSize) && this.unit >= unitSize)) {
 26     throw Error('invalid unit index ' + this.unit);
 27   }
 28   this.texture = gl.createTexture();
 29   this.activate();
 30   this.bindTexture();
 31 };
 32

 34  * The current WebGL context wrapper.

 46 /**
 47  * The actual WebGL texture object.

 49  * @protected
 50  */
 51 cidev.webgl.texture.Texture.prototype.texture;
 52 
 53 /**
 54  * Activates the appropriate texture unit.
 55  * @protected
 56  */
 57 cidev.webgl.texture.Texture.prototype.activate = function() {
 58   this.context.gl.activeTexture(goog.webgl.TEXTURE0 + this.unit);
 59 };
 60 
 61 /**
 62  * Calls the appropriate bind method for the texture.
 63  * @protected
 64  */
 65 cidev.webgl.texture.Texture.prototype.bindTexture = goog.abstractMethod;
