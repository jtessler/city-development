1 /**
  2  * @fileoverview A diffuse and specular lighting vertex and fragment shader.
  3  *
  4  * This class assumes the GLSL shader code includes everything specified in the
  5  * superclasses, plus the following:
  6  *
  7  *   uniform vec3 lightPosition; // The light's position in world coordinates.
  8  *
  9  *   uniform mat4 modelMatrix; // The model transformation matrix.
 10  *   uniform mat4 viewMatrix; // The camera's tranformation matrix.
 11  *   uniform mat4 projMatrix; // The projection matrix.
 12  *
 13  *   uniform vec3 ma; // The mesh's ambient term.
 14  *   uniform vec3 md; // The mesh's diffuse term.
 15  *   uniform vec3 ms; // The mesh's specular term.
 16  *   uniform float shininess;
 17  *
 18  * @author joseph@cs.utexas.edu (Joe Tessler)
 19  */
 20 
 21 goog.provide('cidev.webgl.shader.DiffuseSpecular');
 22 
 23 goog.require('cidev.webgl.shader.TextureProgram');
 24 
 25 goog.require('goog.webgl');
 26 
 27 /**
 28  * @param {!cidev.webgl.Context} context The WebGL context wrapper.
 29  * @param {!goog.vec.Mat4.Float32} matrix Pointer to the scene's model matrix.
 30  * @param {!goog.vec.Vec3.Float32} pos The light's position in world space.
 31  * @constructor
 32  * @extends {cidev.webgl.shader.TextureProgram}
 33  */
 34 cidev.webgl.shader.DiffuseSpecular = function(context, matrix, pos) {
 35   goog.base(
 36       this, context, 'diffuse_specular.vert', 'color_texture.frag', matrix);
 37 
 38   this.lightPositionVector_ = pos;
 39 
 40   var gl = this.context.gl;
 41   this.normal_ = gl.getAttribLocation(this.program, 'normal');
 42   gl.enableVertexAttribArray(this.normal_);
 43 
 44   this.lightPosition_ = gl.getUniformLocation(this.program, 'lightPosition');
 45   this.ma_ = gl.getUniformLocation(this.program, 'ma');
 46   this.md_ = gl.getUniformLocation(this.program, 'md');
 47   this.ms_ = gl.getUniformLocation(this.program, 'ms');
 48   this.shininess_ = gl.getUniformLocation(this.program, 'shininess');
 49 };
 50 goog.inherits(cidev.webgl.shader.DiffuseSpecular,
 51     cidev.webgl.shader.TextureProgram);
 52 
 53 /**
 54  * Sets up the render call, binding vertex attributes, uniforms, etc.
 55  * @inheritDoc
 56  */
 57 cidev.webgl.shader.DiffuseSpecular.prototype.setupRender = function(
 58     mesh, camera, opt_texture) {
 59   goog.base(this, 'setupRender', mesh, camera, opt_texture);
 60 
 61   var gl = this.context.gl;
 62   gl.uniform3fv(this.lightPosition_, this.lightPositionVector_);
 63   gl.uniform3fv(this.ma_, mesh.ma);
 64   gl.uniform3fv(this.md_, mesh.md);
 65   gl.uniform3fv(this.ms_, mesh.ms);
 66   gl.uniform1f(this.shininess_, mesh.shininess);
 67 
 68   mesh.bindNormalBuffer();
 69   this.context.gl.vertexAttribPointer(
 70       this.normal_, 3, goog.webgl.FLOAT, false, 0, 0);
 71 };
