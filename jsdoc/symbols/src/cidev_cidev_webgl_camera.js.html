 16 /**
 17  * @param {!cidev.webgl.Context} context The WebGL context wrapper.
 18  * @constructor
 19  */
 20 cidev.webgl.Camera = function(context) {
 21   /** @type {!cidev.webgl.Context} */
 22   this.context = context;
 23 
 24   /**
 25    * The camera's XYZ coordinates.
 26    * @type {!goog.vec.Vec3.Float32}
 27    */
 28   this.pos = goog.vec.Vec3.createFloat32FromValues(0, 0, 0);
 29 
 30   /**
 31    * The camera's viewing direction (always unit length).
 32    * @type {!goog.vec.Vec3.Float32}
 33    * @private
 34    */
 35   this.dir_ = goog.vec.Vec3.createFloat32FromValues(0, 0, 1);
 36 
 37   /**
 38    * The camera's rotation about the X axis.
 39    * @type {number}
 40    * @private
 41    */
 42   this.pitch_ = Math.PI / 2;
 43 
 44   /**
 45    * The camera's rotation about the Y axis.
 46    * @type {number}
 47    * @private
 48    */
 49   this.yaw_ = Math.PI / 2;
 50 
 51   /**
 52    * The camera's change in rotation about the Y axis.
 53    * @type {number}
 54    * @private
 55    */
 56   this.yawDelta_ = 0;
 57 
 58   /**
 59    * The camera's change in position (frontwards and backwards).
 60    * @type {number}
 61    * @private
 62    */
 63   this.forwardDelta_ = 0;
 64 
 65   /**
 66    * The camera's change in position (right to left).
 67    * @type {number}
 68    * @private
 69    */
 70   this.strafeDelta_ = 0;
 71 
 72   /**
 73    * @type {boolean}
 74    * @private
 75    */
 76   this.mouseDown_ = false;
 77 
 78   /**
 79    * Last seen mouse X position.
 80    * @type {number}
 81    * @private
 82    */
 83   this.mouseX_ = -1;
 84 
 85   /**
 86    * Last seen mouse Y position.
 87    * @type {number}
 88    * @private
 89    */
 90   this.mouseY_ = -1;
 91 
 92   /**
 93    * The view matrix.
 94    * @type {!goog.vec.Mat4.Float32}
 95    */
 96   this.viewMatrix = goog.vec.Mat4.createFloat32();
 97 
 98   /**
 99    * Keep around a constructed vector for various calculations.
100    * @type {!goog.vec.Vec3.Float32}
101    * @private
102    */
103   this.tmpVec3_ = goog.vec.Vec3.createFloat32();
104 
105   goog.events.listen(
106       context.canvas, cidev.webgl.Camera.MOUSE_EVENT_TYPES, this);
107   goog.events.listen(
108       goog.dom.getDocument(), cidev.webgl.Camera.KEY_EVENT_TYPES, this);
109 };
110 
111 /**
112  * Updates the camera's position and rotations based determined by key presses
113  * and mouse movement.
114  * @param {number} dt The elapsed time.
115  */
116 cidev.webgl.Camera.prototype.update = function(dt) {
117   var Vec3 = goog.vec.Vec3;
118 
119   // Determine the change in position based on the camera's viewing direction.
120   var posDelta = Vec3.createFloat32();
121   Vec3.scale(this.dir_, this.forwardDelta_, posDelta);
122   var strafeDir =
123       Vec3.cross(this.dir_, cidev.webgl.Camera.UP_DIR, this.tmpVec3_);
124   Vec3.add(
125       posDelta,
126       Vec3.scale(strafeDir, this.strafeDelta_, this.tmpVec3_),
127       posDelta);
128 
129   // Update the camera's position based on the elapsed time.
130   Vec3.add(
131       this.pos,
132       Vec3.scale(posDelta, dt, this.tmpVec3_),
133       this.pos);
134 
135   // Update the camera's viewing direction based on the elapsed time and the
136   // current rotations.
137   this.yaw_ += dt * this.yawDelta_;
138   Vec3.setFromValues(this.dir_,
139       Math.cos(this.yaw_) * Math.sin(this.pitch_),
140       Math.cos(this.pitch_),
141       Math.sin(this.yaw_) * Math.sin(this.pitch_));
142 
143   goog.vec.Mat4.makeLookAt(
144       this.viewMatrix,
145       this.pos,
146       goog.vec.Vec3.add(this.pos, this.dir_, this.tmpVec3_),
147       cidev.webgl.Camera.UP_DIR);
148 };

150 /**
151  * @param {*} e The event to handle.
152  */
153 cidev.webgl.Camera.prototype.handleEvent = function(e) {
154   var EventType = goog.events.EventType;
155   switch (e.type) {
156     case EventType.KEYUP:
157     case EventType.KEYDOWN:
158         this.handleKeyEvent_(e);
159         break;
160     case EventType.MOUSEDOWN:
161         this.mouseDown_ = e.isMouseActionButton();
162         break;
163     case EventType.MOUSEOUT:
164     case EventType.MOUSEUP:
165         this.mouseDown_ = false;
166         break;
167     case EventType.MOUSEMOVE:
168         this.handleMouseMoveEvent_(e);
169         break;
170     default: throw Error('invalid event type: ' + e.type);
171   }
172 };
173 
174 /**
175  * @param {*} e The key event to handle.
176  * @private
177  */
178 cidev.webgl.Camera.prototype.handleKeyEvent_ = function(e) {
179   var posDelta = 0;
180   var yawDelta = 0;
181   if (e.type == goog.events.EventType.KEYDOWN) {
182     posDelta = cidev.webgl.Camera.KEY_POS_DELTA;
183     yawDelta = cidev.webgl.Camera.KEY_YAW_DELTA;
184   }
185   var KeyCodes = goog.events.KeyCodes;
186   switch (e.keyCode) {
187     case KeyCodes.W: this.forwardDelta_ = posDelta;
188                      break;
189     case KeyCodes.A: this.strafeDelta_ = -posDelta;
190                      break;
191     case KeyCodes.S: this.forwardDelta_ = -posDelta;
192                      break;
193     case KeyCodes.D: this.strafeDelta_ = posDelta;
194                      break;
195     case KeyCodes.Q: this.yawDelta_ = -yawDelta;
196                      break;
197     case KeyCodes.E: this.yawDelta_ = yawDelta;
198                      break;
199   }
200 };

203 /**
204  * @param {*} e The mouse event to handle.
205  * @private
206  */
207 cidev.webgl.Camera.prototype.handleMouseMoveEvent_ = function(e) {
208   if (this.mouseDown_ && this.mouseX_ >= 0 && this.mouseY_ >= 0) {
209     this.yaw_ +=
210         (e.clientX - this.mouseX_) * cidev.webgl.Camera.MOUSE_SENSITIVITY;
211     this.pitch_ +=
212         (this.mouseY_ - e.clientY) * cidev.webgl.Camera.MOUSE_SENSITIVITY;
213   }
214   this.mouseX_ = e.clientX;
215   this.mouseY_ = e.clientY;
216 };
