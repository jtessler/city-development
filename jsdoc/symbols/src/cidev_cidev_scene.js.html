  8 goog.provide('cidev.Scene');

 10 goog.require('cidev.database');

 20 goog.require('goog.vec.Mat4');

 23 /**
 24  * @param {!Element} canvas The canvas container for the scene.
 25  * @constructor
 26  * @implements {goog.fx.anim.Animated}
 27  */
 28 cidev.Scene = function(canvas) {
 29 
 30   /** @type {!cidev.webgl.Context} */
 31   this.context = new cidev.webgl.Context(canvas);
 32 
 33   /** @type {!cidev.webgl.Camera} */
 34   this.camera = new cidev.webgl.Camera(this.context);
 35 
 36   // Setup the initial camera position.
 37   this.camera.pos[0] = 5;
 38   this.camera.pos[1] = 1;
 39 
 40   /** @type {!goog.vec.Mat4.Float32} */
 41   this.modelMatrix = goog.vec.Mat4.createFloat32Identity();
 42 
 43   /**
 44    * Initialize the main shader with a light.
 45    * @type {!cidev.webgl.shader.Program}
 46    */
 47   this.shader = new cidev.webgl.shader.DiffuseSpecular(
 48       this.context,
 49       this.modelMatrix,
 50       goog.vec.Vec3.createFloat32FromValues(-100, 100, 100));
 51 
 52   /** @type {!cidev.webgl.shader.Program} */
 53   this.skybox = new cidev.webgl.shader.Skybox(this.context, this.modelMatrix);
 54 
 55   /** @type {!cidev.webgl.Mesh} */
 56   this.cube = new cidev.webgl.Mesh(this.context, 'cube.obj');
 57 
 58   /** @type {!cidev.webgl.Mesh} */
 59   this.terrain = new cidev.webgl.Mesh(this.context, 'ground.obj');
 60   this.terrain.shininess = 10; // Grass isn't very shiny.
 61 
 62   /** @type {!cidev.webgl.Mesh} */
 63   this.powerPlant = new cidev.webgl.Mesh(this.context, 'power_plant.obj');
 64 
 65   /** @type {!cidev.webgl.Mesh} */
 66   this.residential = new cidev.webgl.Mesh(
 67       this.context, 'residential_building.obj');
 68 
 69   /** @type {!cidev.webgl.Mesh} */
 70   this.school = new cidev.webgl.Mesh(
 71       this.context, 'school.obj');
 72 
 73   /** @type {!cidev.webgl.texture.Texture} */
 74   this.facade = new cidev.webgl.texture.Texture2D(
 75       this.context, 'textures/building_flip.jpg', 0);
 76 
 77   /** @type {!cidev.webgl.texture.Texture} */
 78   this.concrete = new cidev.webgl.texture.Texture2D(
 79       this.context, 'textures/concrete.jpg', 1);
 80 
 81   /** @type {!cidev.webgl.texture.Texture} */
 82   this.grass = new cidev.webgl.texture.Texture2D(
 83       this.context, 'textures/grass.jpg', 2);
 84 
 85   /** @type {!cidev.webgl.texture.Texture} */
 86   this.cubemap = new cidev.webgl.texture.Cubemap(
 87       this.context, 'textures/cubemap', 3);
 88 
 89   /** @type {!cidev.webgl.texture.Texture} */
 90   this.brick = new cidev.webgl.texture.Texture2D(
 91       this.context, 'textures/brick_flip.jpg', 4);
 92 
 93   /**
 94    * Last time seen (used to determine dt).
 95    * @type {number}
 96    * @private
 97    */
 98   this.lastTime_ = 0;
 99 };
100 
101 /**
102  * @inheritDoc
103  */
104 cidev.Scene.prototype.onAnimationFrame = function(now) {
105   this.context.gl.clear(
106       goog.webgl.COLOR_BUFFER_BIT | goog.webgl.DEPTH_BUFFER_BIT);
107 
108   if (this.lastTime_ > 0) {
109     this.camera.update(now - this.lastTime_);
110   }
111   this.lastTime_ = now;
112 
113   this.shader.activate();
114   goog.array.forEach(cidev.database.getAll(),
115       function(building, i, array) { this.renderBuilding_(building); }, this);
116 
117   for (var r = 0; r < 10; r++) {
118     for (var c = 0; c < 10; c++) {
119       goog.vec.Mat4.makeTranslate(this.modelMatrix, 5 * r, 0, 5 * c);
120       goog.vec.Mat4.scale(this.modelMatrix, 5, 5, 5);
121       this.shader.render(this.terrain, this.camera, this.grass);
122     }
123   }
124 
125   this.skybox.activate();
126   goog.vec.Mat4.makeScale(this.modelMatrix, 100, 100, 100);
127   goog.vec.Mat4.translate(this.modelMatrix, -0.25, -0.25, -0.25);
128   this.skybox.render(this.cube, this.camera, this.cubemap);
129 };
130 
131 /**
132  * @param {cidev.model.Building} building The building to render.
133  * @private
134  */
135 cidev.Scene.prototype.renderBuilding_ = function(building) {
136   goog.vec.Mat4.makeTranslate(this.modelMatrix, building.x, 0, building.y);
137   switch (building.getType()) {
138     case cidev.model.BuildingType.POWER_PLANT:
139       var r = building.radius;
140       goog.vec.Mat4.scale(this.modelMatrix, r, r, r);
141       this.shader.render(this.powerPlant, this.camera, this.concrete);
142       break;
143     case cidev.model.BuildingType.RESIDENTIAL:
144       for (var i = 0; i < building.floorCount; i++) {
145         this.shader.render(this.residential, this.camera, this.facade);
146         goog.vec.Mat4.translate(this.modelMatrix, 0, 2, 0);
147       }
148       break;
149     case cidev.model.BuildingType.SCHOOL:
150       this.shader.render(this.school, this.camera, this.brick);
151   }
152 };
