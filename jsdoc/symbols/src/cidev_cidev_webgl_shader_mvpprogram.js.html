1 /**
  2  * @fileoverview The model-view-projection matrix program superclass.
  3  *
  4  * This class assumes the GLSL shader code includes the following:
  5  *
  6  *   attribute vec3 vertexPosition;
  7  *   uniform mat4 modelMatrix; // The model transformation matrix.
  8  *   uniform mat4 viewMatrix; // The camera's tranformation matrix.
  9  *   uniform mat4 projMatrix; // The projection matrix.
 10  *
 11  * @author joseph@cs.utexas.edu (Joe Tessler)
 12  */
 13 
 14 goog.provide('cidev.webgl.shader.MVPProgram');
 15 
 16 goog.require('cidev.webgl.shader.Program');
 17 
 18 /**
 19  * @param {!cidev.webgl.Context} context The WebGL context wrapper.
 20  * @param {string} vs Vertex shader code filename.
 21  * @param {string} fs Fragment shader code filename.
 22  * @param {!goog.vec.Mat4.Float32} matrix Pointer to the scene's model matrix.
 23  * @constructor
 24  * @extends {cidev.webgl.shader.Program}
 25  */
 26 cidev.webgl.shader.MVPProgram = function(context, vs, fs, matrix) {
 27   goog.base(this, context, vs, fs);
 28 
 29   var gl = context.gl;
 30 
 31   this.vertexPosition = gl.getAttribLocation(this.program, 'vertexPosition');
 32   gl.enableVertexAttribArray(this.vertexPosition);
 33 
 34   this.projMatrix = gl.getUniformLocation(this.program, 'projMatrix');
 35   this.viewMatrix = gl.getUniformLocation(this.program, 'viewMatrix');
 36   this.modelMatrix = gl.getUniformLocation(this.program, 'modelMatrix');
 37 
 38   /**
 39    * Pointer to the scene's model matrix (the caller will manipulate it before
 40    * rendering).
 41    * @type {goog.vec.Mat4.Float32}
 42    * @private
 43    */
 44   this.matrix_ = matrix;
 45 };
 46 goog.inherits(cidev.webgl.shader.MVPProgram, cidev.webgl.shader.Program);
 47 
 48 /**
 49  * Sets up the render call, binding vertex attributes, uniforms, etc.
 50  * @inheritDoc
 51  */
 52 cidev.webgl.shader.MVPProgram.prototype.setupRender = function(
 53     mesh, camera, opt_texture) {
 54   goog.base(this, 'setupRender', mesh, camera, opt_texture);
 55 
 56   var gl = this.context.gl;
 57   gl.uniformMatrix4fv(this.projMatrix, false, this.context.projMatrix);
 58   gl.uniformMatrix4fv(this.viewMatrix, false, camera.viewMatrix);
 59   gl.uniformMatrix4fv(this.modelMatrix, false, this.matrix_);
 60 
 61   mesh.bindVertexBuffer();
 62   this.context.gl.vertexAttribPointer(
 63       this.vertexPosition, 3, goog.webgl.FLOAT, false, 0, 0);
 64 };
