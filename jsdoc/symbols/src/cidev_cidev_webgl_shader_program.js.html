  8 goog.provide('cidev.webgl.shader.Program');

 10 goog.require('cidev.generated.glsl');
 11 
 12 goog.require('goog.webgl');

 14 /**
 15  * The super class containing generic shader handling and other utility
 16  * functions.
 17  * @param {!cidev.webgl.Context} context The WebGL context wrapper.
 18  * @param {string} vs Vertex shader code filename.
 19  * @param {string} fs Fragment shader code filename.
 20  * @constructor
 21  */
 22 cidev.webgl.shader.Program = function(context, vs, fs) {
 23   this.context = context;
 24 
 25   if (!goog.isDef(cidev.generated.glsl[vs]) ||
 26       !goog.isDef(cidev.generated.glsl[fs])) {
 27     throw Error('shader does not exist');
 28   }
 29 
 30   var gl = context.gl;
 31   this.program = gl.createProgram();
 32   this.attachShader_(cidev.generated.glsl[vs], goog.webgl.VERTEX_SHADER);
 33   this.attachShader_(cidev.generated.glsl[fs], goog.webgl.FRAGMENT_SHADER);
 34   gl.linkProgram(this.program);
 35 
 36   if (!gl.getProgramParameter(this.program, goog.webgl.LINK_STATUS)) {
 37     throw Error('shader program error: ' + gl.getProgramInfoLog(this.program));
 38   }
 39 };
 40 
 41 /**
 42  * Activates the current program. This caller MUST call this before rendering
 43  * any meshes.
 44  */
 45 cidev.webgl.shader.Program.prototype.activate = function() {
 46   this.context.gl.useProgram(this.program);
 47 };
 48 
 49 /**
 50  * Sets up the render call, binding vertex attributes, uniforms, etc.
 51  * @param {!cidev.webgl.Mesh} mesh The mesh object to draw.
 52  * @param {!cidev.webgl.Camera} camera The view-matrix-wrapping camera.
 53  * @param {!cidev.webgl.texture.Texture=} opt_texture The (optional) texture
 54  *     to apply.
 55  * @protected
 56  */
 57 cidev.webgl.shader.Program.prototype.setupRender = function(
 58     mesh, camera, opt_texture) {
 59   if (goog.isDef(opt_texture) && goog.isDef(this.texture)) {
 60     this.context.gl.uniform1i(this.texture, opt_texture.unit);
 61   }
 62 };
 63 
 64 /**
 65  * Renders the given object.
 66  * @param {!cidev.webgl.Mesh} mesh The mesh object to draw.
 67  * @param {!cidev.webgl.Camera} camera The view-matrix-wrapping camera.
 68  * @param {!cidev.webgl.texture.Texture=} opt_texture The (optional) texture
 69  *     to apply.
 70  */
 71 cidev.webgl.shader.Program.prototype.render = function(
 72     mesh, camera, opt_texture) {
 73   this.setupRender(mesh, camera, opt_texture);
 74   mesh.bindIndexBuffer();
 75   this.context.gl.drawElements(
 76       goog.webgl.TRIANGLES, mesh.indexCount, goog.webgl.UNSIGNED_BYTE, 0);
 77 };
 78 
 79 /**
 80  * Compiles given shader source code into a WebGL shader object, then
 81  * attaches it to the current program.
 82  * @param {string} code GLSL plain text source.
 83  * @param {number} type Shader type, i.e. FRAGMENT_SHADER or VERTEX_SHADER.
 84  * @private
 85  */
 86 cidev.webgl.shader.Program.prototype.attachShader_ = function(code, type) {
 87   var gl = this.context.gl;
 88   var shader = gl.createShader(type);
 89   gl.shaderSource(shader, code);
 90   gl.compileShader(shader);
 91 
 92   if (!gl.getShaderParameter(shader, goog.webgl.COMPILE_STATUS)) {
 93     throw Error('shader error: ' + gl.getShaderInfoLog(shader));
 94   } else if (goog.isNull(shader)) {
 95     throw Error('unknown shader error');
 96   }
 97 
 98   gl.attachShader(this.program, shader);
 99 };
