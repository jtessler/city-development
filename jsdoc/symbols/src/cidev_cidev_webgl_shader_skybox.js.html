1 /**
  2  * @fileoverview A skybox environment mapping vertex and fragment shader.
  3  *
  4  * This class assumes the GLSL shader code includes everything specified in the
  5  * superclasses, plus the following:
  6  *
  7  *   uniform vec3 viewPosition; // The camera's position in world coordinates.
  8  *   uniform samplerCube cubeMapTexture;
  9  *
 10  * @author joseph@cs.utexas.edu (Joe Tessler)
 11  */
 12 
 13 goog.provide('cidev.webgl.shader.Skybox');
 14 
 15 goog.require('cidev.webgl.shader.MVPProgram');
 16 
 17 goog.require('goog.webgl');
 18 
 19 /**
 20  * @param {!cidev.webgl.Context} context The WebGL context wrapper.
 21  * @param {!goog.vec.Mat4.Float32} matrix Pointer to the scene's model matrix.
 22  * @constructor
 23  * @extends {cidev.webgl.shader.MVPProgram}
 24  */
 25 cidev.webgl.shader.Skybox = function(context, matrix) {
 26   goog.base(this, context, 'skybox.vert', 'skybox.frag', matrix);
 27 
 28   var gl = this.context.gl;
 29 
 30   /**
 31    * Location of camera's position in world coordinates.
 32    * @type {WebGLUniformLocation}
 33    * @private
 34    */
 35   this.viewPosition_ = gl.getUniformLocation(this.program, 'viewPosition');
 36 
 37   this.texture = gl.getUniformLocation(this.program, 'cubeMapTexture');
 38 };
 39 goog.inherits(cidev.webgl.shader.Skybox, cidev.webgl.shader.MVPProgram);
 40 
 41 /**
 42  * Sets up the render call, binding vertex attributes, uniforms, etc.
 43  * @inheritDoc
 44  */
 45 cidev.webgl.shader.Skybox.prototype.setupRender = function(
 46     mesh, camera, opt_texture) {
 47   goog.base(this, 'setupRender', mesh, camera, opt_texture);
 48 
 49   this.context.gl.uniform3fv(this.viewPosition_, camera.pos);
 50 };
